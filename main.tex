\includeonly{
    configuration,  % Document configuration / Dokumentkonfiguration
    komaconfig,     % KOMA Script configuration / Konfiguration für KOMA Script
    package,        % List of required packages / Benötigte Pakete
    metadata,       % Document meta data / Dokumentinformationen
    title,          % Title page & / Titelseite
    content,        % Table of contents, lists of figures and tables / Inhalts-, Abbbildungs- und Tabellenverzeichnisse
    foreword,       % Place for a foreword / Vorwort oder Einführung
    document,       % The main document / Das eigentliche Dokument
    appendix,       % The appendices / Die Anhänge
    bibliography,   % Bibliography / Literaturverzeichnis
    index,          % Index

\include{package}
\include{komaconfig}
\include{metadata}

%======================================================================
%   Start of document
%======================================================================
\begin{document}
% This command is only a dummy, but unfortunately required for pageslts
% It is used to enable the roman numbering hack
\pagenumbering{roman}

\include{title}

% Use roman page numbers for table of contents, foreword etc.
\frontmatter

% Render the foreword and the tables of contents
\include{foreword}
\include{content}

% Use arabic page numbers for the main content
\mainmatter
\include{document}
\include{appendix}

% Use roman page numbers for bibliography etc.
\backmatter
\pagenumbering{roman}
\include{bibliography}
\include{index}
\include{authorship}

\end{document}
